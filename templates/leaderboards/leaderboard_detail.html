async function submitAnswer(playerName, answer, score = null, playerId = null) {
  const body = { player_name: playerName, answer: answer };
  if (score !== null) body.score = score;
  if (playerId) body.player_id = playerId;

  const response = await fetch('{{ site_url }}/api/v1/scores/', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'Authorization': 'Bearer {{ leaderboard.api_token }}'
    },
    body: JSON.stringify(body)
  });
  const data = await response.json();
  // Returns: { success: true, message: "Score submitted successfully" }
  // Or error: { success: false, error: "Error message" }
  return data;
}

// Submit answer only (score defaults to 0)
submitAnswer('Player1', 'the answer');

// Submit answer with time score (e.g., milliseconds to solve)
submitAnswer('Player1', 'the answer', 5432);

// With player_id: one entry per player (updates if exists)
submitAnswer('Player1', 'the answer', 5432, 'unique-player-id');

import requests

def submit_answer(player_name, answer, score=None, player_id=None):
    data = {'player_name': player_name, 'answer': answer}
    if score is not None:
        data['score'] = score
    if player_id:
        data['player_id'] = player_id

    response = requests.post(
        '{{ site_url }}/api/v1/scores/',
        headers={
            'Authorization': 'Bearer {{ leaderboard.api_token }}'
        },
        json=data
    )
    # Returns: {'success': True, 'message': 'Score submitted successfully'}
    # Or error: {'success': False, 'error': 'Error message'}
    return response.json()

# Submit answer only (score defaults to 0)
submit_answer('Player1', 'the answer')

# Submit answer with time score (e.g., milliseconds to solve)
submit_answer('Player1', 'the answer', 5432)

# With player_id: one entry per player (updates if exists)
submit_answer('Player1', 'the answer', 5432, 'unique-player-id')
